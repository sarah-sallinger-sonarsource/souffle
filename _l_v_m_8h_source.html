44 /**
45  * Bytecode Interpreter executing a RAM translation unit.
46  *
47  * RAM is transferred into an equivalent Bytecode Representation (LVMCode) by LVMGenerator.
48  * LVM then directly execute the program by interpreting the LVMCode.
49  * The LVMCode will be cached in the memory to save time when repeatedly execution is needed.
50  */

53  LVM(RamTranslationUnit& tUnit) : Interpreter(tUnit) {
54  // Construct mapping from relation Name to RamRelation node in RAM tree.
55  // This will later be used for fast lookup during RamRelationCreate in order to retrieve
56  // minIndexSet from a given relation.
57  visitDepthFirst(*translationUnit.getProgram(), [&](const RamRelation& node) {
58  relNameToNode.insert(std::make_pair(node.getName(), &node));

81  /** Execute the subroutine */
82  void executeSubroutine(const std::string& name, const std::vector<RamDomain>& arguments,
83  std::vector<RamDomain>& returnValues, std::vector<bool>& returnErrors) override {
84  InterpreterContext ctxt;
85  ctxt.setReturnValues(returnValues);
86  ctxt.setReturnErrors(returnErrors);
87  ctxt.setArguments(arguments);
88 
89  if (subroutines.find(name) != subroutines.cend()) {
90  execute(subroutines.at(name), ctxt);
91  } else {
92  // Parse and cache the program
93  LVMGenerator generator(translationUnit.getSymbolTable(),
94  translationUnit.getProgram()->getSubroutine(name), *isa);
95  subroutines.emplace(std::make_pair(name, generator.getCodeStream()));
96  execute(subroutines.at(name), ctxt);
97  }
98  }
