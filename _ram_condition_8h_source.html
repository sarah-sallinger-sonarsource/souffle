399 /**
400  * @brief Convert terms of a conjunction to a list
401  * @param A RAM condition
402  * @param A list of RAM conditions
403  *
404  * Convert a condition of the format C1 /\ C2 /\ ... /\ Cn
405  * to a list {C1, C2, ..., Cn}.
406  */
407 inline std::vector<std::unique_ptr<RamCondition>> toConjunctionList(const RamCondition* condition) {
408  std::vector<std::unique_ptr<RamCondition>> list;
409  std::stack<const RamCondition*> stack;
410  if (condition != nullptr) {
411  stack.push(condition);
412  while (!stack.empty()) {
413  condition = stack.top();
414  stack.pop();
415  if (const auto* ramConj = dynamic_cast<const RamConjunction*>(condition)) {
416  stack.push(&ramConj->getLHS());
417  stack.push(&ramConj->getRHS());
418  } else {
419  list.emplace_back(condition->clone());
420  }
421  }
422  }
423  return list;
424 }

426 /**
427  * @brief Convert list of conditions to a conjunction
428  * @param A list of RAM conditions
429  * @param A RAM condition
430  *
431  * Convert a list {C1, C2, ..., Cn} to a condition of
432  * the format C1 /\ C2 /\ ... /\ Cn.
433  */
434 inline std::unique_ptr<RamCondition> toCondition(const std::vector<const RamCondition*>& list) {
435  std::unique_ptr<RamCondition> result;
436  for (const RamCondition* cur : list) {
437  if (result == nullptr) {
438  result = std::unique_ptr<RamCondition>(cur->clone());
439  } else {
440  result = std::make_unique<RamConjunction>(
441  std::move(result), std::unique_ptr<RamCondition>(cur->clone()));
442  }
443  }
444  return result;
445 }
