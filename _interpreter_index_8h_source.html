35  struct comparator {

39  comparator(LexOrder order) : order(std::move(order)) {}

41  /* comparison function */
42  int operator()(const RamDomain* x, const RamDomain* y) const {
43  for (int i : order) {
44  if (x[i] < y[i]) {
45  return -1;
46  }
47  if (x[i] > y[i]) {
48  return 1;
49  }
50  }
51  return 0;
52  }

54  /* less comparison */
55  bool less(const RamDomain* x, const RamDomain* y) const {
56  return operator()(x, y) < 0;
57  }
58 
59  /* equal comparison */
60  bool equal(const RamDomain* x, const RamDomain* y) const {
61  for (int i : order) {
62  if (x[i] != y[i]) {
63  return false;
64  }
65  }
66  return true;
67  }
68  };
