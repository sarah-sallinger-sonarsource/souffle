33 bool ExpandFilterTransformer::expandFilters(RamProgram& program) {

35  bool changed = false;
36 
37  visitDepthFirst(program, [&](const RamQuery& query) {
38  std::function<std::unique_ptr<RamNode>(std::unique_ptr<RamNode>)> filterRewriter =
39  [&](std::unique_ptr<RamNode> node) -> std::unique_ptr<RamNode> {
40  if (const RamFilter* filter = dynamic_cast<RamFilter*>(node.get())) {
41  const RamCondition* condition = &filter->getCondition();
42  std::vector<std::unique_ptr<RamCondition>> conditionList = toConjunctionList(condition);
43  if (conditionList.size() > 1) {
44  changed = true;
45  std::vector<std::unique_ptr<RamFilter>> filters;
46  for (auto iter = conditionList.rbegin(); iter != conditionList.rend(); ++iter) {
47  auto& cond = *iter;
48  auto tempCond = cond->clone();
49  if (filters.empty()) {
50  filters.emplace_back(std::make_unique<RamFilter>(
51  std::unique_ptr<RamCondition>(std::move(tempCond)),
52  std::unique_ptr<RamOperation>(filter->getOperation().clone())));
53  } else {
54  filters.emplace_back(std::make_unique<RamFilter>(
55  std::unique_ptr<RamCondition>(std::move(tempCond)),
56  std::move(filters.back())));
57  }
58  }
59  node = std::move(filters.back());
60  }
61  }
62  node->apply(makeLambdaRamMapper(filterRewriter));
63  return node;
64  };
65  const_cast<RamQuery*>(&query)->apply(makeLambdaRamMapper(filterRewriter));

108 bool HoistConditionsTransformer::hoistConditions(RamProgram& program) {
109  // flag to determine whether the RAM program has changed
110  bool changed = false;
111 
112  // helper for collecting conditions from filter operations
113  auto addCondition = [](std::unique_ptr<RamCondition> condition,
114  RamCondition* c) -> std::unique_ptr<RamCondition> {
115  if (condition == nullptr) {
116  return std::unique_ptr<RamCondition>(c);
117  } else {
118  return std::make_unique<RamConjunction>(std::move(condition), std::unique_ptr<RamCondition>(c));
119  }
120  };
121 
122  // insert a new filter
123  auto insertFilter = [](RamOperation* op, std::unique_ptr<RamCondition>& condition) {
124  op->apply(makeLambdaRamMapper([&](std::unique_ptr<RamNode> node) -> std::unique_ptr<RamNode> {
125  if (nullptr != dynamic_cast<RamOperation*>(node.get())) {
126  return std::make_unique<RamFilter>(std::move(condition),
127  std::unique_ptr<RamOperation>(dynamic_cast<RamOperation*>(node.release())));
128  }
129  return node;
130  }));
131  };
132 
133  // hoist conditions to the most outer scope if they
134  // don't depend on RamSearches
135  visitDepthFirst(program, [&](const RamQuery& query) {
136  std::unique_ptr<RamCondition> newCondition;
137  std::function<std::unique_ptr<RamNode>(std::unique_ptr<RamNode>)> filterRewriter =
138  [&](std::unique_ptr<RamNode> node) -> std::unique_ptr<RamNode> {
139  if (auto* filter = dynamic_cast<RamFilter*>(node.get())) {
140  const RamCondition& condition = filter->getCondition();
141  // if filter condition is independent of any RamSearch,
142  // delete the filter operation and collect condition
143  if (rla->getLevel(&condition) == -1) {
144  changed = true;
145  newCondition = addCondition(std::move(newCondition), condition.clone());
146  node->apply(makeLambdaRamMapper(filterRewriter));
147  return std::unique_ptr<RamOperation>(filter->getOperation().clone());
148  }
149  }
150  node->apply(makeLambdaRamMapper(filterRewriter));
151  return node;
152  };
153  const_cast<RamQuery*>(&query)->apply(makeLambdaRamMapper(filterRewriter));
154  if (newCondition != nullptr) {
155  // insert new filter operation at outer-most level of the query
156  changed = true;
157  insertFilter((RamOperation*)&query, newCondition);
158  }
159  });
160 
161  // hoist conditions for each RamSearch operation
162  visitDepthFirst(program, [&](const RamSearch& search) {
163  std::unique_ptr<RamCondition> newCondition;
164  std::function<std::unique_ptr<RamNode>(std::unique_ptr<RamNode>)> filterRewriter =
165  [&](std::unique_ptr<RamNode> node) -> std::unique_ptr<RamNode> {
166  if (auto* filter = dynamic_cast<RamFilter*>(node.get())) {
167  const RamCondition& condition = filter->getCondition();
168  // if filter condition matches level of RamSearch,
169  // delete the filter operation and collect condition
170  if (rla->getLevel(&condition) == search.getTupleId()) {
171  changed = true;
172  newCondition = addCondition(std::move(newCondition), condition.clone());
173  node->apply(makeLambdaRamMapper(filterRewriter));
174  return std::unique_ptr<RamOperation>(filter->getOperation().clone());
175  }
176  }
177  node->apply(makeLambdaRamMapper(filterRewriter));
178  return node;
179  };
180  const_cast<RamSearch*>(&search)->apply(makeLambdaRamMapper(filterRewriter));
181  if (newCondition != nullptr) {
182  // insert new filter operation after the search operation
183  changed = true;
184  insertFilter((RamOperation*)&search, newCondition);
185  }
186  });

205  return std::unique_ptr<RamExpression>(lhs->clone());

215  std::vector<std::unique_ptr<RamCondition>> conditionList, int identifier) {
216  // Remaining conditions which cannot be handled by an index
217  std::unique_ptr<RamCondition> condition;
218  auto addCondition = [&](std::unique_ptr<RamCondition> c) {
219  if (condition != nullptr) {
220  condition = std::make_unique<RamConjunction>(std::move(condition), std::move(c));
221  } else {
222  condition = std::move(c);
223  }
224  };
225 
226  // Build query pattern and remaining condition
227  for (auto& cond : conditionList) {
228  size_t element = 0;
229  if (std::unique_ptr<RamExpression> value = getExpression(cond.get(), element, identifier)) {
230  if (queryPattern[element] == nullptr) {
231  indexable = true;
232  queryPattern[element] = std::move(value);

234  // TODO: This case is a recursive case introducing a new filter operation

240  std::unique_ptr<RamExpression>(queryPattern[element]->clone())));

242  } else {

291  if (const auto* filter = dynamic_cast<const RamFilter*>(&iscan->getOperation())) {
292  const RamRelation& rel = iscan->getRelation();

295  bool indexable = false;

297  queryPattern, indexable, toConjunctionList(&filter->getCondition()), identifier);
298  if (indexable) {

300  const auto prevPattern = iscan->getRangePattern();
301  auto addCondition = [&](std::unique_ptr<RamCondition> c) {
302  if (condition != nullptr) {
303  condition = std::make_unique<RamConjunction>(std::move(condition), std::move(c));
304  } else {

306  }
307  };
308  for (std::size_t i = 0; i < rel.getArity(); i++) {
309  if (prevPattern[i] != nullptr) {
310  if (queryPattern[i] == nullptr) {

320  }
321  }

557 bool HoistAggregateTransformer::hoistAggregate(RamProgram& program) {
558  // flag to determine whether the RAM program has changed
559  bool changed = false;
560 
561  // hoist aggregate if doesn't depend on anything
562  visitDepthFirst(program, [&](const RamQuery& query) {
563  int currLevel = -1;
564  int oldLevel = -1;
565  bool hoist = false;
566  bool newIndex = false;
567 
568  // Fields for new aggregate
569  AggregateFunction newFun;
570  const RamRelation* newRef;
571  RamExpression* newExp;
572  RamCondition* newCond;
573  std::vector<RamExpression*> newPattern;
574 
575  // Tracking aggregates seen to determine whether to
576  // hoist or not
577  std::vector<int> aggIds;
578 
579  // Removing an aggregate that can be hoisted if it exists
580  std::function<std::unique_ptr<RamNode>(std::unique_ptr<RamNode>)> aggRemover =
581  [&](std::unique_ptr<RamNode> node) -> std::unique_ptr<RamNode> {
582  // Checking that there exists an aggregate to hoist
583  // If so, remove the aggregate from the loop nest
584  if (const RamAggregate* agg = dynamic_cast<RamAggregate*>(node.get())) {
585  aggIds.push_back(agg->getTupleId());

587  if (!hoist && rla->getLevel(agg) < agg->getTupleId() - 1) {
588  // If all searches between the rla->getLevel(agg) and agg
589  // are aggregates, then we do not transform

623  break;

637  newPattern = agg->getRangePattern();
638  }

641  node->apply(makeLambdaRamMapper(aggRemover));

643  };

645  const_cast<RamQuery*>(&query)->apply(makeLambdaRamMapper(aggRemover));
646 
647  bool added = false;
648

650  std::function<std::unique_ptr<RamNode>(std::unique_ptr<RamNode>)> aggAdder =
651  [&](std::unique_ptr<RamNode> node) -> std::unique_ptr<RamNode> {
652  if (auto* search = dynamic_cast<RamSearch*>(node.get())) {
653  if (!added && search->getTupleId() == currLevel - 1) {
654  added = true;

656  // Finding the operation right before the RamSearch
657  // with tupleId = aggregate's tupleId - 1

659  op = dynamic_cast<RamNestedOperation*>(&op->getOperation());
660  }
661

665  for (const RamExpression* cur : newPattern) {
666  if (nullptr != cur) {

668  } else {
669  queryPattern.push_back(nullptr);
670  }
671  }

678  std::unique_ptr<RamExpression>(newExp),
679  std::unique_ptr<RamCondition>(newCond), std::move(queryPattern), oldLevel);

684

686  std::unique_ptr<RamOperation>(op->getOperation().clone()), newFun,
687  std::make_unique<RamRelationReference>(newRef),
688  std::unique_ptr<RamExpression>(newExp),
689  std::unique_ptr<RamCondition>(newCond), oldLevel);
690  }
691  }
692  }
693  node->apply(makeLambdaRamMapper(aggAdder));
694  return node;
695  };

698  [&](std::unique_ptr<RamNode> node) -> std::unique_ptr<RamNode> {
699  if (nullptr != dynamic_cast<RamOperation*>(node.get())) {
700  if (newIndex) {

706  } else {

708  }
709  }
710

712  std::unique_ptr<RamOperation>(dynamic_cast<RamOperation*>(node.release())),
713  newFun, std::make_unique<RamRelationReference>(newRef),
714  std::unique_ptr<RamExpression>(newExp), std::unique_ptr<RamCondition>(newCond),
715  std::move(queryPattern), oldLevel);
716  } else {
717  // RamAggregate
718  return std::make_unique<RamAggregate>(
719  std::unique_ptr<RamOperation>(dynamic_cast<RamOperation*>(node.release())),
720  newFun, std::make_unique<RamRelationReference>(newRef),
721  std::unique_ptr<RamExpression>(newExp), std::unique_ptr<RamCondition>(newCond),
722  oldLevel);
723  }
724  }
725  return node;
726  };
727 
728  if (hoist) {
729  if (currLevel < 1) {
730  // insert at the top of the loop nest
731  const_cast<RamQuery*>(&query)->apply(makeLambdaRamMapper(aggTopAdder));
732  } else {
733  const_cast<RamQuery*>(&query)->apply(makeLambdaRamMapper(aggAdder));
734  }
735  }
736  });
737
