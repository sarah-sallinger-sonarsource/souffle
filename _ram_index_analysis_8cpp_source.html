57 bool MaxMatching::bfSearch() {
58  SearchSignature u;
59  std::queue<SearchSignature> bfQueue;
60  // Build layers
61  for (auto& it : graph) {
62  if (getMatch(it.first) == RIA_NIL) {
63  distance[it.first] = 0;
64  bfQueue.push(it.first);
65  } else {
66  distance[it.first] = RIA_INF;
67  }
68  }
69  distance[RIA_NIL] = RIA_INF;
70  while (!bfQueue.empty()) {
71  u = bfQueue.front();
72  bfQueue.pop();
73  assert(u != RIA_NIL);
74  const Edges& children = graph[u];
75  for (auto it : children) {
76  SearchSignature mv = getMatch(it);
77  if (getDistance(mv) == RIA_INF) {
78  distance[mv] = getDistance(u) + 1;
79  if (mv != RIA_NIL) {
80  bfQueue.push(mv);
81  }
82  }
83  }
84  }
85  return (getDistance(0) != RIA_INF);
86 }

118 void MinIndexSelection::solve() {
119  // map the keys in the key set to lexicographical order
120  if (searches.empty()) {
121  return;
122  }
123 
124  // check whether one of the naive indexers should be used
125  // two conditions: either set by environment or relation is a hash map
126  static const char ENV_NAIVE_INDEX[] = "SOUFFLE_USE_NAIVE_INDEX";
127  if (std::getenv(ENV_NAIVE_INDEX)) {
128  static bool first = true;
129  // print a warning - only the first time
130  if (first) {
131  std::cout << "WARNING: auto index selection disabled, naive indexes are utilized!!\n";
132  first = false;
133  }
134 
135  // every search pattern gets its naive index
136  for (SearchSignature cur : searches) {
137  // obtain order
138  LexOrder order;
139  SearchSignature mask = cur;
140  for (int i = 0; mask != 0; i++) {
141  if (!(1 << i & mask)) {
142  continue;
143  }
144  order.push_back(i);
145  // clear bit
146  mask &= ~(1 << i);
147  }
148 
149  // add new order
150  orders.push_back(order);
151 
152  // register pseudo chain
153  chainToOrder.push_back(Chain());
154  chainToOrder.back().insert(cur);
155  }
156 
157  return;
158  }
159 
160  // Construct the matching poblem
161  for (auto search : searches) {
162  // For this node check if other nodes are strict subsets
163  for (auto itt : searches) {
164  if (isStrictSubset(search, itt)) {
165  matching.addEdge(search, toB(itt));
166  }
167  }
168  }
169 
170  // Perform the Hopcroft-Karp on the graph and receive matchings (mapped A->B and B->A)
171  // Assume: alg.calculate is not called on an empty graph
172  assert(!searches.empty());
173  const MaxMatching::Matchings& matchings = matching.solve();
174 
175  // Extract the chains given the nodes and matchings
176  const ChainOrderMap chains = getChainsFromMatching(matchings, searches);
177 
178  // Should never get no chains back as we never call calculate on an empty graph
179  assert(!chains.empty());
180 
181  for (const auto& chain : chains) {
182  std::vector<int> ids;
183  SearchSignature initDelta = *(chain.begin());
184  insertIndex(ids, initDelta);
185 
186  for (auto iit = chain.begin(); next(iit) != chain.end(); ++iit) {
187  SearchSignature delta = *(next(iit)) - *iit;
188  insertIndex(ids, delta);
189  }
190 
191  assert(!ids.empty());
192 
193  orders.push_back(ids);
194  }
195 
196  // Construct the matching poblem
197  for (auto search : searches) {
198  int idx = map(search);
199  size_t l = card(search);
200  SearchSignature k = 0;
201  for (size_t i = 0; i < l; i++) {
202  k = k + (1 << (orders[idx][i]));
203  }
204  assert(k == search && "incorrect lexicographical order");
205  }
206 }

208 MinIndexSelection::Chain MinIndexSelection::getChain(
209  const SearchSignature umn, const MaxMatching::Matchings& match) {
210  SearchSignature start = umn; // start at an unmateched node
211  Chain chain;
212  // given an unmapped node from set A we follow it from set B until it cannot be matched from B
213  // if not mateched from B then umn is a chain
214  //
215  // Assume : no circular mappings, i.e. a in A -> b in B -> ........ -> a in A is not allowed.
216  // Given this, the loop will terminate
217  while (true) {
218  auto mit = match.find(toB(start)); // we start from B side
219  chain.insert(start);
220 
221  if (mit == match.end()) {
222  return chain;
223  }
224 
225  SearchSignature a = mit->second;
226  chain.insert(a);
227  start = a;
228  }
229 }

231 const MinIndexSelection::ChainOrderMap MinIndexSelection::getChainsFromMatching(
232  const MaxMatching::Matchings& match, const SearchSet& nodes) {
233  assert(!nodes.empty());
234 
235  // Get all unmatched nodes from A
236  const SearchSet& umKeys = getUnmatchedKeys(match, nodes);
237 
238  // Case: if no unmatched nodes then we have an anti-chain
239  if (umKeys.empty()) {
240  for (auto node : nodes) {
241  std::set<SearchSignature> a;
242  a.insert(node);
243  chainToOrder.push_back(a);
244  return chainToOrder;
245  }
246  }
247 
248  assert(!umKeys.empty());
249 
250  // A worklist of used nodes
251  SearchSet usedKeys;
252 
253  // Case: nodes < umKeys or if nodes == umKeys then anti chain - this is handled by this loop
254  for (auto umKey : umKeys) {
255  Chain c = getChain(umKey, match);
256  assert(!c.empty());
257  chainToOrder.push_back(c);
258  }
259 
260  assert(!chainToOrder.empty());
261 
262  return chainToOrder;
263 }

265 void RamIndexAnalysis::run(const RamTranslationUnit& translationUnit) {
266  // After complete:
267  // 1. All relations should have at least one index (for full-order search).
268  // 2. Two relations involved in a swap operation will have same set of indices.
269  // 3. A 0-arity relation will have only one index where LexOrder is defined as empty. A comparator using
270  // an empty order should regard all elements as equal and therefore only allow one arbitrary tuple to be
271  // inserted.
272  //
273  // TODO:
274  // 0-arity relation in a provenance program still need to be revisited.
275 
276  // visit all nodes to collect searches of each relation
277  visitDepthFirst(*translationUnit.getProgram(), [&](const RamNode& node) {
278  if (const auto* indexSearch = dynamic_cast<const RamIndexRelationSearch*>(&node)) {
279  MinIndexSelection& indexes = getIndexes(indexSearch->getRelation());
280  indexes.addSearch(getSearchSignature(indexSearch));
281  } else if (const auto* exists = dynamic_cast<const RamExistenceCheck*>(&node)) {
282  MinIndexSelection& indexes = getIndexes(exists->getRelation());
283  indexes.addSearch(getSearchSignature(exists));
284  } else if (const auto* provExists = dynamic_cast<const RamProvenanceExistenceCheck*>(&node)) {
285  MinIndexSelection& indexes = getIndexes(provExists->getRelation());
286  indexes.addSearch(getSearchSignature(provExists));
287  } else if (const auto* ramRel = dynamic_cast<const RamRelation*>(&node)) {
288  MinIndexSelection& indexes = getIndexes(*ramRel);
289  indexes.addSearch(getSearchSignature(ramRel));
290  }
291  });
292 
293  // A swap happen between rel A and rel B indicates A should include all indices of B, vice versa.
294  visitDepthFirst(*translationUnit.getProgram(), [&](const RamSwap& swap) {
295  // Note: this naive approach will not work if there exists chain or cyclic swapping.
296  // e.g. swap(relA, relB) swap(relB, relC) swap(relC, relA)
297  // One need to keep merging the search set until a fixed point where no more index is introduced
298  // in any of the relation in a complete iteration.
299  //
300  // Currently RAM does not have such situation.
301  const RamRelation& relA = swap.getFirstRelation();
302  const RamRelation& relB = swap.getSecondRelation();
303 
304  MinIndexSelection& indexesA = getIndexes(relA);
305  MinIndexSelection& indexesB = getIndexes(relB);
306  // Add all searchSignature of A into B
307  for (const auto& signature : indexesA.getSearches()) {
308  indexesB.addSearch(signature);
309  }
310 
311  // Add all searchSignature of B into A
312  for (const auto& signature : indexesB.getSearches()) {
313  indexesA.addSearch(signature);
314  }
315  });
316 
317  // find optimal indexes for relations
318  for (auto& cur : minIndexCover) {
319  MinIndexSelection& indexes = cur.second;
320  indexes.solve();
321  }
322 
323  // Only case where indexSet is still empty is when relation has arity == 0
324  for (auto& cur : minIndexCover) {
325  MinIndexSelection& indexes = cur.second;
326  if (indexes.getAllOrders().empty()) {
327  indexes.insertDefaultTotalIndex(0);
328  }
329  }
330 }
